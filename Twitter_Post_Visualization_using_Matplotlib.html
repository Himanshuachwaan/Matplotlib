fig, ax = plt.subplots(figsize=(7,5), facecolor='#999999', dpi=200)
ax.set_facecolor('#999999') # for color inside box behind bars

(df.to_pandas()
.plot.barh(x='Manufacturer',y='Cars_Registered', legend=False, width = 0.85, edgecolor='black', ax=ax,color = 'yellow'))
ax.tick_params(axis='both', labelcolor='#ffffff', length=-0.8)# changing labels color on both axis
ax.set_ylabel('')# no label on y axis
ax.spines[['left','top','right']].set_visible(False)#border lines removed
ax.spines['bottom'].set_linewidth(2)# increasing the width of bottom line
ax.spines['bottom'].set_color('#ffffff')# setting color to the bottom line
fig.suptitle('USA New Ev Registrations\n2023(Jan - Jul)', fontsize=16, color='white', weight=1000, y=.75)# Title
ax.xaxis.set_major_formatter(ticker.FuncFormatter(lambda x, pos: f'{int(x):,}'))# to make values on x axis more easily understandable
ax.text(x=.12, y=.025, s='Created by Himanshu Chaudhary | Source:World of Statistics, @stats_feed, Joram_Mutenge',
        transform=fig.transFigure, ha='left', fontsize=8, weight=300, color='white')# adding my name and sources from where i learned and took the data from
